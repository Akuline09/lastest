local function getAllTeamNames()
    local names = {}
    for _, team in ipairs(Teams:GetTeams()) do
        table.insert(names, team.Name)
    end
    return names
end

local function refreshTeamList()
    local names = getAllTeamNames()
    for _, name in ipairs(names) do
        if getgenv().AdvTeamCheck_Teams[name] == nil then
            getgenv().AdvTeamCheck_Teams[name] = true
        end
        if getgenv().AdvAuraTeamCheck_Teams[name] == nil then
            getgenv().AdvAuraTeamCheck_Teams[name] = true
        end
    end
    return names
end

local function refreshNPCs()
    npcCache = {}
    for _, obj in ipairs(workspace:GetDescendants()) do
        if obj:IsA("Model") and obj ~= LocalPlayer.Character then
            local hum = obj:FindFirstChildOfClass("Humanoid")
            local hrp = obj:FindFirstChild("HumanoidRootPart")
            if hum and hrp then
                local isPC = false
                for _, p in ipairs(Players:GetPlayers()) do
                    if p.Character == obj then isPC = true; break end
                end
                if not isPC then table.insert(npcCache, obj) end
            end
        end
    end
end

-- NPC HOSTILE / FRIENDLY
-- ============================================================
local function isNPCTargetingPlayer(model)
    local hum = model:FindFirstChildOfClass("Humanoid")
    if not hum then return false end

    local targeting = false
    pcall(function()
        if hum.Target and hum.Target:IsDescendantOf(workspace) then
            for _, p in ipairs(Players:GetPlayers()) do
                if p.Character and (hum.Target == p.Character or hum.Target:IsDescendantOf(p.Character)) then
                    targeting = true
                    return
                end
            end
        end
    end)
    if targeting then return true end

    for _, desc in ipairs(model:GetDescendants()) do
        if desc:IsA("ObjectValue") and desc.Value then
            local n = desc.Name:lower()
            if n:find("target") or n:find("attack") or n:find("victim") or n:find("aggro") then
                for _, p in ipairs(Players:GetPlayers()) do
                    local chr = p.Character
                    if chr and (desc.Value == chr or desc.Value:IsDescendantOf(chr)) then
                        return true
                    end
                end
            end
        end
    end

    local hrp = model:FindFirstChild("HumanoidRootPart")
    local myHrp = LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("HumanoidRootPart")
    if hrp and myHrp then
        local dist = (hrp.Position - myHrp.Position).Magnitude
        if dist < 50 then
            local lookDir = hrp.CFrame.LookVector
            local toPlayer = (myHrp.Position - hrp.Position).Unit
            local dot = lookDir:Dot(toPlayer)
            if dot > 0.7 then
                if hum and hum.MoveDirection.Magnitude > 0 then
                    local moveDir = hum.MoveDirection.Unit
                    local dot2 = moveDir:Dot(toPlayer)
                    if dot2 > 0.5 then
                        return true
                    end
                end
            end
        end
    end

    return false
end

local function getNPCAlign(model)
    if isNPCTargetingPlayer(model) then
        return "hostile"
    end

    for _, a in ipairs({"Hostile","IsHostile","Enemy","IsEnemy","Aggressive","hostile","enemy"}) do
        if model:GetAttribute(a) == true then return "hostile" end
    end
    for _, a in ipairs({"Friendly","IsFriendly","Ally","IsAlly","friendly","ally"}) do
        if model:GetAttribute(a) == true then return "friendly" end
    end

    for _, desc in ipairs(model:GetDescendants()) do
        if desc:IsA("BoolValue") then
            local n = desc.Name:lower()
            if (n=="hostile" or n=="enemy" or n=="aggressive") and desc.Value then return "hostile" end
            if (n=="friendly" or n=="ally") and desc.Value then return "friendly" end
        end
    end

    return "unknown"
end

local function shouldTargetNPC(model)
    if not getgenv().TargetNPC then return false end
    if not model or not model.Parent then return false end
    local hum = model:FindFirstChildOfClass("Humanoid")
    if not hum or hum.Health <= 0 then return false end

    local align = getNPCAlign(model)
    if getgenv().NPCHostileOnly and align ~= "hostile" then return false end
    if getgenv().NPCFriendlyOnly and align ~= "friendly" then return false end

    return true
end

local function getNPCColor(model)
    local align = getNPCAlign(model)
    if align == "hostile"  then return Color3.fromRGB(255, 60, 60) end
    if align == "friendly" then return Color3.fromRGB(60, 220, 60) end
    return Color3.fromRGB(255, 160, 0)
end

-- ============================================================
-- TARGET LOGIC
-- ============================================================
local function shouldTarget(v)
    if v == LocalPlayer then return false end
    if not getgenv().TargetPlayers then return false end

    if getgenv().AdvTeamCheck_Enabled then
        if v.Team then
            local teamName = v.Team.Name
            return getgenv().AdvTeamCheck_Teams[teamName] == true
        end
        return true
    end

    if getgenv().HeadTeamCheck and LocalPlayer.Team == v.Team then return false end
    if getgenv().HeadFriendOnly and LocalPlayer:IsFriendsWith(v.UserId) then return false end
    return true
end

local function shouldTargetTorso(v)
    if v == LocalPlayer then return false end
    if not getgenv().TargetPlayers then return false end

    if getgenv().AdvTeamCheck_Enabled then
        if v.Team then
            local teamName = v.Team.Name
            return getgenv().AdvTeamCheck_Teams[teamName] == true
        end
        return true
    end

    if getgenv().TorsoTeamCheck and LocalPlayer.Team == v.Team then return false end
    if getgenv().TorsoFriendOnly and LocalPlayer:IsFriendsWith(v.UserId) then return false end
    return true
end

local function auraTarget(v)
    if v == LocalPlayer then return false end

    if getgenv().AdvAuraTeamCheck_Enabled then
        if v.Team then
            local teamName = v.Team.Name
            return getgenv().AdvAuraTeamCheck_Teams[teamName] == true
        end
        return true
    end

    if getgenv().AuraTeamCheck and LocalPlayer.Team == v.Team then return false end
    if getgenv().AuraFriendOnly and LocalPlayer:IsFriendsWith(v.UserId) then return false end
    return true
end

local function getTeamColor(v)
    if v.Team then return v.Team.TeamColor.Color end
    return Color3.fromRGB(255, 50, 50)
end

-- ============================================================
-- FAKE PART MANAGER - FIX TRANSPARENCY & COLLISION
-- ============================================================
local function safeTransparency(t)
    return math.clamp(t, 0, 0.99)
end

local function createFakeHead(targetKey, realHead)
    if fakeHeads[targetKey] and fakeHeads[targetKey].Parent then
        return fakeHeads[targetKey]
    end

    if fakeHeads[targetKey] then
        pcall(function() fakeHeads[targetKey]:Destroy() end)
    end

    local fake = Instance.new("Part")
    fake.Name = "Head"
    fake.Shape = Enum.PartType.Ball
    fake.Size = Vector3.new(getgenv().HeadSize, getgenv().HeadSize, getgenv().HeadSize)
    fake.Transparency = safeTransparency(getgenv().HeadTransparency)
    fake.CanCollide = false
    fake.CanTouch = true
    fake.CanQuery = true
    fake.Massless = true
    fake.Anchored = false
    fake.Color = realHead.Color
    fake.Material = realHead.Material
    fake.TopSurface = Enum.SurfaceType.Smooth
    fake.BottomSurface = Enum.SurfaceType.Smooth
    fake.Parent = realHead.Parent

    local weld = Instance.new("Weld")
    weld.Part0 = fake
    weld.Part1 = realHead
    weld.C0 = CFrame.new(0, 0, 0)
    weld.Parent = fake

    fakeHeads[targetKey] = fake
    return fake
end

local function createFakeTorso(targetKey, realTorso)
    if fakeTorsos[targetKey] and fakeTorsos[targetKey].Parent then
        return fakeTorsos[targetKey]

end

local function removeFakeHead(targetKey)
    local fake = fakeHeads[targetKey]
    if fake then
        pcall(function() fake:Destroy() end)
        fakeHeads[targetKey] = nil
    end
end

local function removeFakeTorso(targetKey)
    local fake = fakeTorsos[targetKey]
    if fake then
        pcall(function() fake:Destroy() end)
        fakeTorsos[targetKey] = nil
    end
end

local function clearAllFakeHeads()
    for k, fake in pairs(fakeHeads) do
        pcall(function() fake:Destroy() end)
    end
    fakeHeads = {}
end

local function clearAllFakeTorsos()
    for k, fake in pairs(fakeTorsos) do
        pcall(function() fake:Destroy() end)
    end
    fakeTorsos = {}
end

local function getTorsoFromCharacter(char)
    return char:FindFirstChild("UpperTorso") or char:FindFirstChild("Torso") or char:FindFirstChild("LowerTorso") or char:FindFirstChild("HumanoidRootPart")
end

local function applyHeads()
    local type1 = getgenv().HeadSizeType == 1
    local type2 = getgenv().HeadSizeType == 2
    local hitboxOn = getgenv().HeadHitboxOn
    local torsoOn = getgenv().TorsoHitboxOn

    if torsoOn then
        for k in pairs(fakeHeads) do
            removeFakeHead(k)
        end
        for _, v in ipairs(Players:GetPlayers()) do
            if v.Character then
                local head = v.Character:FindFirstChild("Head")
                if head then
                    head.Size = Vector3.new(2, 1, 1)
                    head.Transparency = 0
                    head.Massless = false
                    head.CanCollide = false
                end
            end
        end
        for _, model in ipairs(npcCache) do
            if model and model.Parent then
                local head = model:FindFirstChild("Head")
                if head then
                    head.Size = Vector3.new(2, 1, 1)
                    head.Transparency = 0
                    head.Massless = false
                    head.CanCollide = false
                end
            end
        end
        return
    end

    -- Players
    for _, v in ipairs(Players:GetPlayers()) do
        if v == LocalPlayer or not v.Character then
            removeFakeHead(v)
        else
            local head = v.Character:FindFirstChild("Head")
            if not head then
                removeFakeHead(v)
            else
                local shouldHit = hitboxOn and shouldTarget(v)
                if type1 then
                    removeFakeHead(v)
                    if shouldHit then
                        head.Size = Vector3.new(getgenv().HeadSize, getgenv().HeadSize, getgenv().HeadSize)
                        head.Transparency = safeTransparency(getgenv().HeadTransparency)
                        head.Massless = true
                        head.CanCollide = false
                    else
                        head.Size = Vector3.new(2, 1, 1)
                        head.Transparency = 0
                        head.Massless = false
                        head.CanCollide = false
                    end
                elseif type2 then
                    head.Size = Vector3.new(2, 1, 1)
                    head.Transparency = 0
                    head.Massless = false
                    head.CanCollide = false
                    if shouldHit then
                        local fake = createFakeHead(v, head)
                        fake.Size = Vector3.new(getgenv().HeadSize, getgenv().HeadSize, getgenv().HeadSize)
                        fake.Transparency = safeTransparency(getgenv().HeadTransparency)
                    else
                        removeFakeHead(v)
                    end
                else
                    removeFakeHead(v)
                    head.Size = Vector3.new(2, 1, 1)
                    head.Transparency = 0
                    head.Massless = false
                    head.CanCollide = false
                end
            end
        end
    end

    -- NPC
    for _, model in ipairs(npcCache) do
        if not model or not model.Parent then
            removeFakeHead(model)
        else
            local head = model:FindFirstChild("Head")
            if not head then
                removeFakeHead(model)
            else
                local shouldHit = hitboxOn and shouldTargetNPC(model)
                if type1 then
                    removeFakeHead(model)
                    if shouldHit then
                        head.Size = Vector3.new(getgenv().HeadSize, getgenv().HeadSize, getgenv().HeadSize)
                        head.Transparency = safeTransparency(getgenv().HeadTransparency)
                        head.Massless = true
                        head.CanCollide = false
                    else
                        head.Size = Vector3.new(2, 1, 1)
                        head.Transparency = 0
                        head.Massless = false
                        head.CanCollide = false
                    end
                elseif type2 then
                    head.Size = Vector3.new(2, 1, 1)
                    head.Transparency = 0
                    head.Massless = false
                    head.CanCollide = false
                    if shouldHit then
                        local fake = createFakeHead(model, head)
                        if fake then
                            fake.Size = Vector3.new(getgenv().HeadSize, getgenv().HeadSize, getgenv().HeadSize)
                            fake.Transparency = safeTransparency(getgenv().HeadTransparency)
                        end
                    else
                        removeFakeHead(model)
                    end
                else
                    removeFakeHead(model)
                    head.Size = Vector3.new(2, 1, 1)
                    head.Transparency = 0
                    head.Massless = false
                    head.CanCollide = false
                end
            end
        end
    end
end

local function applyTorsoHitbox()
    if not getgenv().TorsoHitboxOn then
        clearAllFakeTorsos()
        return
    end

    for _, v in ipairs(Players:GetPlayers()) do
        if v == LocalPlayer or not v.Character then
            removeFakeTorso(v)
        else
            local torso = getTorsoFromCharacter(v.Character)
            if not torso then
                removeFakeTorso(v)
            else
                local shouldHit = shouldTargetTorso(v)
                if shouldHit then
                    local fake = createFakeTorso(v, torso)
                    fake.Size = Vector3.new(getgenv().TorsoSize, getgenv().TorsoSize, getgenv().TorsoSize)
                    fake.Transparency = safeTransparency(getgenv().TorsoTransparency)
                else
                    removeFakeTorso(v)
                end
            end
        end
    end

    for _, model in ipairs(npcCache) do
        if not model or not model.Parent then
            removeFakeTorso(model)
        else
            local torso = getTorsoFromCharacter(model)
            if not torso then
                removeFakeTorso(model)
            else
                local shouldHit = shouldTargetNPC(model)
                if shouldHit then
                    local fake = createFakeTorso(model, torso)
                    fake.Size = Vector3.new(getgenv().TorsoSize, getgenv().TorsoSize, getgenv().TorsoSize)
                    fake.Transparency = safeTransparency(getgenv().TorsoTransparency)
                else
                    removeFakeTorso(model)
                end
            end
        end
    end
end

local function onPlayerCharacterAdded(player, character)
    task.delay(0.5, function()
        if not character or not character.Parent then return end
        if getgenv().HeadHitboxOn and getgenv().HeadSizeType == 2 and shouldTarget(player) then
            local head = character:FindFirstChild("Head")
            if head then
                head.Size = Vector3.new(2, 1, 1)
                head.Transparency = 0
                head.Massless = false
                head.CanCollide = false
                createFakeHead(player, head)
            end
        end
        if getgenv().TorsoHitboxOn and shouldTargetTorso(player) then
            local torso = getTorsoFromCharacter(character)
            if torso then
                createFakeTorso(player, torso)
            end
        end
    end)
end

for _, v in ipairs(Players:GetPlayers()) do
    if v ~= LocalPlayer then
        playerCharacters[v] = v.Character
        v.CharacterAdded:Connect(function(char)
            playerCharacters[v] = char
            onPlayerCharacterAdded(v, char)
        end)
    end
end

Players.PlayerAdded:Connect(function(v)
    if v == LocalPlayer then return end
    playerCharacters[v] = v.Character
    v.CharacterAdded:Connect(function(char)
        playerCharacters[v] = char
        onPlayerCharacterAdded(v, char)
    end)
    if v.Character then
        onPlayerCharacterAdded(v, v.Character)
    end
end)

Players.PlayerRemoving:Connect(function(plr)
    removeFakeHead(plr)
    removeFakeTorso(plr)
    playerCharacters[plr] = nil
end)

RunService.RenderStepped:Connect(function()
    if not getgenv().AuraOn then return end
    local chr = LocalPlayer.Character; if not chr then return end
    local inRange = false

    for _, v in ipairs(Players:GetPlayers()) do
        if auraTarget(v) then
            local vc = v.Character; if vc then
                local hum = vc:FindFirstChildOfClass("Humanoid")
                local hrp = vc:FindFirstChild("HumanoidRootPart")
                if hum and hrp and hum.Health > 0 then
                    if LocalPlayer:DistanceFromCharacter(hrp.Position) <= getgenv().AuraRange then
                        inRange = true
                        if getgenv().AuraMode == "tool" then
                            local tool = chr:FindFirstChildOfClass("Tool")
                            if tool and tool:FindFirstChild("Handle") then
                                tool:Activate()
                                for _, part in next, vc:GetChildren() do
                                    if part:IsA("BasePart") then
                                        firetouchinterest(tool.Handle, part, 0)
                                        firetouchinterest(tool.Handle, part, 1)
                                    end
                                end
                            end
                        end
                    end
                end
            end
        end
    end

    if getgenv().TargetNPC then
        for _, model in ipairs(npcCache) do
            if shouldTargetNPC(model) then
                local hrp = model:FindFirstChild("HumanoidRootPart")
                if hrp and LocalPlayer:DistanceFromCharacter(hrp.Position) <= getgenv().AuraRange then
                    inRange = true
                    if getgenv().AuraMode == "tool" then
                        local tool = chr:FindFirstChildOfClass("Tool")
                        if tool and tool:FindFirstChild("Handle") then
                            tool:Activate()
                            for _, part in next, model:GetChildren() do
                                if part:IsA("BasePart") then
                                    firetouchinterest(tool.Handle, part, 0)
                                    firetouchinterest(tool.Handle, part, 1)
                                end
                            end
                        end
                    end
                end
            end
        end
    end

    if getgenv().AuraMode == "m1" and inRange then
        local tool = chr:FindFirstChildOfClass("Tool")
        if isMobile then
            if tool then pcall(function() tool:Activate() end) end
        else
            pcall(function() mouse1click() end)
        end
    end
end)

local espData = {}

local function clearESP(key)
    local d = espData[key]; if not d then return end
    for _, obj in pairs(d) do
        pcall(function()
            if typeof(obj) == "Instance" then obj:Destroy() else obj:Remove() end
        end)
    end
    espData[key] = nil
end

local function clearAll()
    for k in pairs(espData) do clearESP(k) end
end

local function newText(col)
    local t = Drawing.new("Text")
    t.Visible = true; t.Color = col; t.Size = 13
    t.Font = Drawing.Fonts.UI; t.Outline = true
    t.OutlineColor = Color3.new(0, 0, 0); t.Center = true
    return t
end

local function newLine(col)
    local l = Drawing.new("Line")
    l.Visible = true; l.Color = col; l.Thickness = 1.2; l.Transparency = 1
    return l
end

local function getBounds(model)
    local hrp  = model:FindFirstChild("HumanoidRootPart")
    local head = model:FindFirstChild("Head")
    if not hrp or not head then return end
    local top, onT = Camera:WorldToViewportPoint(head.Position + Vector3.new(0, head.Size.Y * 0.5, 0))
    local bot, onB = Camera:WorldToViewportPoint(hrp.Position - Vector3.new(0, 3, 0))
    if not onT or not onB or top.Z <= 0 then return end
    return top, bot
end

local function drawESP(key, model, col, nameStr, dist)
    if not espData[key] then espData[key] = {} end
    local d = espData[key]

    if getgenv().ESP_Outline then
        if not d.hl then
            local hl = Instance.new("Highlight")
            hl.FillTransparency    = 1
            hl.OutlineTransparency = 0
            hl.DepthMode           = Enum.HighlightDepthMode.AlwaysOnTop
            hl.Parent              = workspace
            d.hl = hl
        end
        d.hl.Adornee      = model
        d.hl.OutlineColor = col
    elseif d.hl then
        d.hl:Destroy(); d.hl = nil
    end

    local top, bot = getBounds(model)
    if not top then
        for _, k in pairs({"tr","nm","di","hpt"}) do
            if d[k] then pcall(function() d[k]:Remove() end); d[k] = nil end
        end
        return
    end

    local vp  = Camera.ViewportSize
    local tb  = Vector2.new(vp.X * 0.5, vp.Y)
    local cx  = top.X
    local y1  = top.Y
    local y2  = bot.Y

    if getgenv().ESP_Tracer then
        if not d.tr then d.tr = newLine(col) end
        d.tr.Color = col; d.tr.From = tb; d.tr.To = Vector2.new(cx, y2); d.tr.Visible = true
    elseif d.tr then d.tr.Visible = false end

    local infoY = y1 - 16
    if getgenv().ESP_Info then
        if not d.nm then d.nm = newText(col) end
        if not d.di then d.di = newText(Color3.new(1, 1, 1)) end
        d.nm.Color = col; d.nm.Text = nameStr
        d.nm.Position = Vector2.new(cx, infoY); d.nm.Visible = true
        d.di.Text = dist .. " m"
        d.di.Position = Vector2.new(cx, infoY + 13); d.di.Visible = true
    elseif d.nm then
        d.nm.Visible = false
        if d.di then d.di.Visible = false end
    end

    if getgenv().ESP_HP then
        local hum   = model:FindFirstChildOfClass("Humanoid")
        local hp    = hum and math.floor(hum.Health)    or 0
        local maxHp = hum and math.max(math.floor(hum.MaxHealth), 1) or 100
        local pct   = math.floor(hp / maxHp * 100)
        local r     = math.floor(255 * (1 - pct / 100))
        local g     = math.floor(255 * (pct / 100))
        local hpCol = Color3.fromRGB(r, g, 0)
        if not d.hpt then d.hpt = newText(hpCol) end
        d.hpt.Color    = hpCol
        d.hpt.Text     = hp .. " HP (" .. pct .. "%)"
        local hpY = getgenv().ESP_Info and (infoY + 26) or (infoY + 13)
        d.hpt.Position = Vector2.new(cx, hpY); d.hpt.Visible = true

            end

            if shouldTargetNPC(model) then

            local isP = false
            for _, p in ipairs(Players:GetPlayers()) do if p == k then isP = true; break end end
            if not isP and not active[k] then clearESP(k) end
        end
    end

    for k in pairs(espData) do
        if typeof(k) == "Instance" and k:IsA("Player") and not k.Parent then clearESP(k) end
    end
end)

local keybindActions = {}
local currentKeybinds = {
    Hitbox = Enum.KeyCode.F,
    Aura   = Enum.KeyCode.G,
    GUI    = Enum.KeyCode.Insert,
}

local function registerKeybind(actionName, newKeyCode, actionFunc)
    for key, info in pairs(keybindActions) do
        if info.name == actionName then
            keybindActions[key] = nil
            break
        end
    end
    if newKeyCode then
        keybindActions[newKeyCode] = { name = actionName, func = actionFunc }
    end
    currentKeybinds[actionName] = newKeyCode
end

UIS.InputBegan:Connect(function(input, gameProcessed)
    if gameProcessed then return end
    if input.UserInputType ~= Enum.UserInputType.Keyboard then return end
    local info = keybindActions[input.KeyCode]
    if info then
        pcall(info.func)
    end
end)

local function parseKeyCode(keyName)
    if not keyName or keyName == "" then return nil end
    keyName = tostring(keyName):gsub("%s+", "")
    local keyMap = {
        ["Semicolon"] = Enum.KeyCode.Semicolon, [";"] = Enum.KeyCode.Semicolon,
        ["Comma"] = Enum.KeyCode.Comma, [","] = Enum.KeyCode.Comma,
        ["Period"] = Enum.KeyCode.Period, ["."] = Enum.KeyCode.Period,
        ["Slash"] = Enum.KeyCode.Slash, ["/"] = Enum.KeyCode.Slash,
        ["Quote"] = Enum.KeyCode.Quote, ["\'"] = Enum.KeyCode.Quote,
        ["BackSlash"] = Enum.KeyCode.BackSlash, ["\\"] = Enum.KeyCode.BackSlash,
        ["LeftBracket"] = Enum.KeyCode.LeftBracket, ["["] = Enum.KeyCode.LeftBracket,
        ["RightBracket"] = Enum.KeyCode.RightBracket, ["]"] = Enum.KeyCode.RightBracket,
        ["Minus"] = Enum.KeyCode.Minus, ["-"] = Enum.KeyCode.Minus,
        ["Equals"] = Enum.KeyCode.Equals, ["="] = Enum.KeyCode.Equals,
        ["Grave"] = Enum.KeyCode.Grave, ["`"] = Enum.KeyCode.Grave,
        ["RightShift"] = Enum.KeyCode.RightShift, ["LeftShift"] = Enum.KeyCode.LeftShift,
        ["RightControl"] = Enum.KeyCode.RightControl, ["LeftControl"] = Enum.KeyCode.LeftControl,
        ["RightAlt"] = Enum.KeyCode.RightAlt, ["LeftAlt"] = Enum.KeyCode.LeftAlt,
        ["Insert"] = Enum.KeyCode.Insert, ["Delete"] = Enum.KeyCode.Delete,
        ["Home"] = Enum.KeyCode.Home, ["End"] = Enum.KeyCode.End,
        ["PageUp"] = Enum.KeyCode.PageUp, ["PageDown"] = Enum.KeyCode.PageDown,
        ["Return"] = Enum.KeyCode.Return, ["Enter"] = Enum.KeyCode.Return,
        ["Backspace"] = Enum.KeyCode.Backspace, ["Tab"] = Enum.KeyCode.Tab,
        ["CapsLock"] = Enum.KeyCode.CapsLock, ["Escape"] = Enum.KeyCode.Escape,
        ["Space"] = Enum.KeyCode.Space,
        ["Up"] = Enum.KeyCode.Up, ["Down"] = Enum.KeyCode.Down,
        ["Left"] = Enum.KeyCode.Left, ["Right"] = Enum.KeyCode.Right,
    }
    if keyMap[keyName] then return keyMap[keyName] end
    local ok, result = pcall(function() return Enum.KeyCode[keyName] end)
    if ok and result then return result end
    return nil
end
